    class CoolexType
    {
        public List<CoolexType> Block;
        public CoolexLex.Type Type;
        public string Value;
		public int Line;

        public CoolexType(CoolexLex.Type type, string value, int line)
        {
            Type = type;
            Value = value;
			Line = line;
            Block = null;
        }
    }
