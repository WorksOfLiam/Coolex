class CoolexToken {
  /**
   * 
   * @param {Type} type 
   * @param {string} value 
   * @param {number} index 
   */
  constructor(type, value, index) {
    this.Type = Number(type);
    this.Value = value;
    this.Index = index;
    this.Block = null;
  }
}

class ParseError 
{
	constructor(index, text) {
		this.Index = index;
		this.Text = text;
	}
}
